def showInverseList(lista):
	newList = []

	for indice in range(0, len(lista)):
		newList.insert(0, lista[indice])
	
	print newList

if __name__ == '__main__':
	lista = [10,9,8,7]
	showInverseList(lista)

Montar uma lista de três números com valores que o usuário passa pelo terminal 

def listByUser():

	userList = []

	print 'Insira três números'

	while len(userList) < 3:
		
		element = raw_input('Entre com um número: ')
		
		if element.isdigit():
			userList.append(element)
		
		else:
			print 'Elemento não pode ser inserido'
			element = raw_input('Entre com um número: ')

	print userList

if __name__ == '__main__':
	listByUser()
